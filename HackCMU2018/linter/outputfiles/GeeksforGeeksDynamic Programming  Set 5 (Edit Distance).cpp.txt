#include <bits/stdc++.h>
using namespace std;

int dp[102][102];
string a,b;
int solve(int i,int j)
{
    if(i==0)
        return j;
    if(j==0)
        return i;
    if(dp[i][j]!=-1)
        return dp[i][j];
    if(a[i-1]==b[j-1])
        dp[i][j]=solve(i-1,j-1);
    else
        dp[i][j]=1+min(solve(i-1,j),min(solve(i,j-1),solve(i-1,j-1)));
    return dp[i][j];
}
