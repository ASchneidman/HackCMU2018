long long n;
long long tree[4][1002][1002];
int maxx,maxy;
void update(int x,int y,long long val,int i)
{
    int y1=y;
    while(x<=maxx)
    {
        y1=y;
        while(y1<=maxy)
        {
            tree[i][x][y1]+=val;
            y1+=(y1 & -y1);
        }
        x+=(x&-x);
    }
}
long long submat(int x,int y,int i)
{
        int yt;
        long long sum=0;
        while(x>0)
        {
            yt=y;
            while(yt>0)
            {
                sum+=tree[i][x][yt];
                yt-=(yt&-yt);
            }
            x-=(x&-x);
        }
        return sum;
}
