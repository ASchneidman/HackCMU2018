	int value()

		return 5;

	const double x = 100;
	double value()

		return 2 * x;

namespace ns3
{
	void display();
	class Test
	{
	public:
		void Display()
		{
			cout << "ns3::Test::Display() " << endl;
		}
	};

void ns3::display()
{
	cout << "Display " << endl;
}

// nested namespace
namespace outer
{
	int x = 10;
	namespace inner
	{
		int z = x;
	}
}
