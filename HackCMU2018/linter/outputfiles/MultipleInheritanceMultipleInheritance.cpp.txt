class A
{

	A()
	{
		cout << "A constructor called " << endl;
	}
	~A()
	{
		cout << "A Destructor called " << endl;
	}
};

{
public:
	B()
	{
		cout << "B constructor called " << endl;
	}
	~B()
	{
		cout << "B Destructor called " << endl;
	}
};

{
public:
	C()
	{
		cout << "C constructor called " << endl;

		cout << "C Destructor called " << endl;
	}
};
